cmake_minimum_required(VERSION 3.15)\nproject(CNDS-Complex-Network-Dynamics-Simulator \n    VERSION 1.0.0\n    DESCRIPTION "Visualização 3D interativa de dinâmica em rede complexa"\n    LANGUAGES C\n)\n\n# Configurações gerais\nset(CMAKE_C_STANDARD 99)\nset(CMAKE_C_STANDARD_REQUIRED ON)\nset(CMAKE_C_FLAGS_DEBUG "-g -O0 -Wall -Wextra -Wpedantic")\nset(CMAKE_C_FLAGS_RELEASE "-O3 -Wall -Wextra")\n\n# Diretórios\nset(PROJECT_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}")\nset(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")\n\n# Encontrar dependências\nfind_package(OpenGL REQUIRED)\nfind_package(GLUT REQUIRED)\n\n# Incluir diretórios\ninclude_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)\ninclude_directories(${OPENGL_INCLUDE_DIR})\ninclude_directories(${GLUT_INCLUDE_DIR})\n\n# Arquivos fonte\nset(SOURCES\n    src/main.c\n    src/network.c\n    src/graphics.c\n)\n\n# Executável principal\nadd_executable(complex-network-sim ${SOURCES})\n\n# Link bibliotecas\ntarget_link_libraries(complex-network-sim\n    PRIVATE\n    ${OPENGL_LIBRARIES}\n    ${GLUT_LIBRARY}\n    m  # Biblioteca matemática\n)\n\n# Propriedades do target\nset_target_properties(complex-network-sim PROPERTIES\n    VERSION ${PROJECT_VERSION}\n    SOVERSION 1\n)\n\n# Opções de compilação por SO\nif(WIN32)\n    target_link_options(complex-network-sim PRIVATE -mwindows)\nendif()\n\n# Testes (preparação para futuro)\nenable_testing()\n\n# Instalação (opcional)\ninstall(TARGETS complex-network-sim DESTINATION bin)\ninstall(DIRECTORY include/ DESTINATION include/complex-network-sim)\n\n# Mensagem de build\nmessage(STATUS "Build type: ${CMAKE_BUILD_TYPE}")\nmessage(STATUS "OpenGL found: ${OPENGL_FOUND}")\nmessage(STATUS "GLUT found: ${GLUT_FOUND}")\n
